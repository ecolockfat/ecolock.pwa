/**
 * ========================================
    * GESTIÓN DE ALMACENAMIENTO LOCAL
 * ========================================
 * 
 * Propósito: Abstraer el uso de localStorage con:
 * - Prefijo para evitar colisiones
 * - Manejo de errores (localStorage puede fallar si está lleno)
 * - Serialización/deserialización automática de objetos
 * - Validación de datos antes de guardar
 */

const Storage = {
    // ========================================
    // MÉTODOS BÁSICOS
    // ========================================
    
    /**
     * Guarda un valor en localStorage.
     * @param {string} key - Clave (sin prefijo, se agrega automáticamente)
     * @param {any} value - Valor a guardar (se serializa si es objeto)
     * @returns {boolean} - True si se guardó correctamente
     */
    set(key, value) {
        try {
            const fullKey = CONFIG.STORAGE_PREFIX + key;
            const serialized = JSON.stringify(value);
            localStorage.setItem(fullKey, serialized);
            return true;
        } catch (error) {
            // localStorage puede fallar por:
            // 1. QuotaExceededError (almacenamiento lleno)
            // 2. SecurityError (navegación privada)
            console.error('Error guardando en localStorage:', error);
            UI.showToast('⚠️ No se pudo guardar la configuración', 'warning');
            return false;
        }
    },
    
    /**
     * Obtiene un valor de localStorage.
     * @param {string} key - Clave (sin prefijo)
     * @param {any} defaultValue - Valor por defecto si no existe
     * @returns {any} - Valor deserializado o defaultValue
     */
    get(key, defaultValue = null) {
        try {
            const fullKey = CONFIG.STORAGE_PREFIX + key;
            const item = localStorage.getItem(fullKey);
            
            if (item === null) return defaultValue;
            
            // Intentamos parsear como JSON, si falla devolvemos el string crudo
            try {
                return JSON.parse(item);
            } catch {
                return item;
            }
        } catch (error) {
            console.error('Error leyendo localStorage:', error);
            return defaultValue;
        }
    },
    
    /**
     * Elimina una clave de localStorage.
     * @param {string} key - Clave a eliminar
     */
    remove(key) {
        try {
            const fullKey = CONFIG.STORAGE_PREFIX + key;
            localStorage.removeItem(fullKey);
        } catch (error) {
            console.error('Error eliminando de localStorage:', error);
        }
    },
    
    /**
     * Limpia TODOS los datos de EcoLock del localStorage.
     * ¡Cuidado! Esto borra configuración y logs.
     */
    clearAll() {
        try {
            // Solo borramos las claves con nuestro prefijo
            const keysToRemove = [];
            for (let i = 0; i < localStorage.length; i++) {
                const key = localStorage.key(i);
                if (key && key.startsWith(CONFIG.STORAGE_PREFIX)) {
                    keysToRemove.push(key);
                }
            }
            keysToRemove.forEach(key => localStorage.removeItem(key));
            return true;
        } catch (error) {
            console.error('Error limpiando localStorage:', error);
            return false;
        }
    },
    
    // ========================================
    // MÉTODOS ESPECÍFICOS DE LA APP
    // ========================================
    
    /**
     * Guarda la IP del ESP32 con validación.
     * @param {string} ip - Dirección IP
     * @returns {boolean}
     */
    saveIP(ip) {
        if (!Utils.isValidIP(ip)) {
            throw ErrorHandler.createValidationError('La IP ingresada no es válida. Formato: 192.168.1.100');
        }
        return this.set(CONFIG.STORAGE_KEYS.IP, ip.trim());
    },
    
    /**
     * Obtiene la IP guardada del ESP32.
     * @returns {string|null}
     */
    getIP() {
        return this.get(CONFIG.STORAGE_KEYS.IP);
    },
    
    /**
     * Guarda el intervalo de polling.
     * @param {number} interval - Milisegundos
     */
    savePollingInterval(interval) {
        const validIntervals = Object.values(CONFIG.POLLING_INTERVALS);
        if (!validIntervals.includes(interval)) {
            interval = CONFIG.DEFAULT_POLLING_INTERVAL;
        }
        return this.set(CONFIG.STORAGE_KEYS.POLLING, interval);
    },
    
    /**
     * Obtiene el intervalo de polling guardado.
     * @returns {number}
     */
    getPollingInterval() {
        return this.get(CONFIG.STORAGE_KEYS.POLLING, CONFIG.DEFAULT_POLLING_INTERVAL);
    },
    
    /**
     * Guarda preferencia de modo oscuro.
     * @param {boolean} isDark
     */
    saveDarkMode(isDark) {
        return this.set(CONFIG.STORAGE_KEYS.DARK_MODE, isDark);
    },
    
    /**
     * Obtiene preferencia de modo oscuro.
     * @returns {boolean}
     */
    getDarkMode() {
        return this.get(CONFIG.STORAGE_KEYS.DARK_MODE, true); // Default: modo oscuro
    },
    
    // ========================================
    // MANEJO DE ACCIONES PENDIENTES (OFFLINE)
    // ========================================
    
    /**
     * Agrega una acción a la cola de pendientes.
     * Se usa cuando no hay conexión con el ESP32.
     * @param {string} action - 'lock' o 'unlock'
     */
    addPendingAction(action) {
        const pending = this.get(CONFIG.STORAGE_KEYS.PENDING_ACTIONS) || [];
        pending.push({
            action: action,
            timestamp: Date.now()
        });
        this.set(CONFIG.STORAGE_KEYS.PENDING_ACTIONS, pending);
    },
    
    /**
     * Obtiene todas las acciones pendientes.
     * @returns {Array}
     */
    getPendingActions() {
        return this.get(CONFIG.STORAGE_KEYS.PENDING_ACTIONS) || [];
    },
    
    /**
     * Limpia las acciones pendientes (después de sincronizar).
     */
    clearPendingActions() {
        this.remove(CONFIG.STORAGE_KEYS.PENDING_ACTIONS);
    },
    
    /**
     * Elimina una acción pendiente específica.
     * @param {number} index - Índice en el array
     */
    removePendingAction(index) {
        const pending = this.getPendingActions();
        pending.splice(index, 1);
        this.set(CONFIG.STORAGE_KEYS.PENDING_ACTIONS, pending);
    }
};

window.Storage = Storage;
